## 配置

### 基础站点配置

```typescript
// docs/athen.config.ts
import { defineConfig } from 'athen';

export default defineConfig({
  title: '我的文档',
  description: '我的项目的综合指南',
  base: '/',
  
  // 启用搜索
  search: {
    provider: 'flex'
  },
  
  themeConfig: {
    // 导航
    nav: [
      { text: '首页', link: '/zh/' },
      { text: '指南', link: '/zh/guide/getting-started' },
      { text: 'API', link: '/zh/api/reference' }
    ],
    
    // 侧边栏
    sidebar: {
      '/zh/guide/': [
        {
          text: '开始使用',
          items: [
            { text: '介绍', link: '/zh/guide/getting-started' },
            { text: '配置', link: '/zh/guide/configuration' }
          ]
        }
      ]
    },
    
    // 社交链接
    socialLinks: [
      { icon: 'github', mode: 'link', content: 'https://github.com/your-repo' }
    ],
    
    // 页脚
    footer: {
      message: '基于 MIT 许可证发布',
      copyright: 'Copyright © 2024 您的名字'
    }
  }
});
```

## 部署

### 静态托管

部署到任何静态托管服务：

```bash
# 构建站点
npm run build

# 将 dist/ 文件夹部署到您的托管服务
```

### GitHub Pages

```yaml
# .github/workflows/deploy.yml
name: 部署到 GitHub Pages

on:
  push:
    branches: [main]

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - uses: actions/setup-node@v3
        with:
          node-version: 18
      - run: npm ci
      - run: npm run build
      - uses: peaceiris/actions-gh-pages@v3
        with:
          github_token: ${{ secrets.GITHUB_TOKEN }}
          publish_dir: ./dist
```
